MINIHELL

command execution

	minishell> "" (ou) ''
		error no stderr
		exit code = 127


	minishell> "        " (ou) '       '
		erro no stderr
		exit code = 127

	minishell> "''" (ou) '""'
		erro no stderr
		exit code = 127

	minishell> $key
		nada deve ser feito na tela
		exit code = 0

	minishell> ls
		comando é invocado
		exit code = 0

	minishell> /bin/ls ou /sbin/ping 
		comando é invocado
		exit code = 0

	
	minishell> chmod -x ./minishell
	minishell> ./minishell
		erro no stderror
		exit code = 126

	minishell> "echo Hello World"
		erro no stderror
		exit code = 126

cd	
	cd (puro)
		deve voltar para o diretorio em HOME
		exit code = 0

	cd .
		deve ficar na mesma pasta
		PWD e OLDPWD ficam com o valor do diretorio atual.

export

	export (puro)
		deve exibir o env
		exit code = 0

	export key=val key1=val1 key2=val2
		env | grep key (deve exibir todas as variavéis acima)
		exit code = 0

	export 1=a
		err no stderr
		exit code = 1
